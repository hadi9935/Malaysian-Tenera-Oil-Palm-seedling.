<script>
    const ctx = document.getElementById('growthChart').getContext('2d');
    let chart;

    function initChart() {
        const labels = Array.from({length: 25}, (_, i) => i);
        const data = labels.map(x => (0.08 * x * x + 1.5 * x + 10).toFixed(2));

        chart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Seedling Height (cm)',
                    data: data,
                    borderColor: '#2e7d32',
                    backgroundColor: 'rgba(46, 125, 50, 0.1)',
                    fill: true,
                    tension: 0.4,
                    pointRadius: 0
                }]
            },
            options: {
                plugins: { legend: { display: false } },
                scales: { y: { beginAtZero: true } }
            }
        });
    }

    function updateAll() {
        const x = document.getElementById('ageInput').value;
        const height = (0.08 * x * x + 1.5 * x + 10).toFixed(2);
        const rate = (0.16 * x + 1.5).toFixed(2);

        document.getElementById('displayMonth').innerText = x;
        document.getElementById('displayHeight').innerText = height;

        const statusLabel = document.getElementById('statusLabel');
        const fieldNote = document.getElementById('fieldNote');

        if (x < 12) {
            statusLabel.innerText = "Phase: Main Nursery";
            statusLabel.className = "status-pill nursery";
            fieldNote.innerText = "Seedling requires intensive NPK application and shade monitoring.";
        } else {
            statusLabel.innerText = "Phase: Field Ready";
            statusLabel.className = "status-pill field";
            fieldNote.innerText = "Optimal physiological vigor reached. Suitable for plantation transplanting.";
        }
    }

    initChart();
    updateAll();
</script>
